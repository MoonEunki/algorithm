# 음료수 얼려 먹기

```javascript
function solution(arr) {
  const dx = [-1, 0, 1, 0];
  const dy = [0, 1, 0, -1];
  let answer = 0;
  for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr[0].length; j++) {
      if (arr[i][j] == 0) {
        DFS(i, j);
        answer++;
      }
    }
  }
  return answer;

  //상,우,하,좌
  function DFS(x, y) {
    arr[x][y] = 1;
    for (let i = 0; i < 4; i++) {
      if (
        x + dx[i] < arr.length &&
        x + dx[i] >= 0 &&
        y + dy[i] < arr[0].length &&
        y + dy[i] >= 0
      ) {
        if (arr[x + dx[i]][y + dy[i]] == 0) {
          DFS(x + dx[i], y + dy[i]);
        }
      }
    }
  }
}
```
